select name_cust , sum(sele_rest) from tab_deble group by id_cus , name_cust order by id_cus



SELECT d.name_cust ,d.order_id ,d.sele_rest , d.dat , s.sele_rest_paid , s.cash ,s.bank,s.num_bok,s.dat
from tab_deble d JOIN tab_deble_paid s
ON d.id_cus = s.id_cus where d.name_cust = 'احمد جاد' ORDER BY d.id ASC 


SELECT o.ored_id , o.ored_date , o.user_add ,
f.order_id , f.ITEM_NAME_AR , f.ITEM_COUNT , f.ITEM_PRIC , f.totalAmount ,
t.name_cust , t.order_id , t.sele_rest , t.dat , 
p.p_name_cust , p.p_order_id , p.sele_rest_paid ,p.cash , p.bank , p.num_bok , p.p_dat
from itemorder o JOIN itemOrderDeail f on o.ored_id = f.order_id
JOIN tab_deble t on f.order_id = t.order_id 
JOIN tab_deble_paid p on f.order_id =  p.p_order_id where  t.sele_rest > 0
